  match n:
    case 0n:
      0n
    case 1n:
      0n
    case 2n + p:
      1n + div2(p)

  match n:
    case 0n:
      0n
    case 1n + 0n:
      1n
    case 1n + (1n + p):
      2n + complex(p)
